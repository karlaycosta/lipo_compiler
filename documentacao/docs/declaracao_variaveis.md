# Com Tipo Explícito:

* inteiro idade = 25;

* real altura = 1.75;

* texto nome = "Maria";

* logico ativo = verdadeiro;



#  Com Inferência de Tipo:

* var idade = 25;           // inferido como inteiro

* var altura = 1.75;        // inferido como real

* var nome = "Maria";       // inferido como texto

* var ativo = verdadeiro;   // inferido como lógico

# Sem Inicialização:

* inteiro contador;
* real media;

* texto resultado;
